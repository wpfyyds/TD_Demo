## 3. Unity Shader特效实现方案

### 3.1 敌人死亡特效Shader

```shader
// PixelExplosion.shader
Shader "Custom/PixelExplosion"
{
    Properties
    {
        _MainTex ("Texture", 2D) = "white" {}
        _PixelSize ("Pixel Size", Range(1, 10)) = 4
        _Color ("Color", Color) = (1, 0.5, 0, 1)
        _FadeSpeed ("Fade Speed", Range(0.1, 5)) = 2
    }
    SubShader
    {
        Tags { "Queue"="Transparent" "RenderType"="Transparent" }
        LOD 100
        Blend SrcAlpha OneMinusSrcAlpha
        
        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            
            #include "UnityCG.cginc"
            
            struct appdata
            {
                float4 vertex : POSITION;
                float2 uv : TEXCOORD0;
            };
            
            struct v2f
            {
                float2 uv : TEXCOORD0;
                float4 vertex : SV_POSITION;
                float3 worldPos : TEXCOORD1;
            };
            
            sampler2D _MainTex;
            float4 _MainTex_ST;
            float _PixelSize;
            float4 _Color;
            float _FadeSpeed;
            
            v2f vert (appdata v)
            {
                v2f o;
                o.vertex = UnityObjectToClipPos(v.vertex);
                o.uv = TRANSFORM_TEX(v.uv, _MainTex);
                o.worldPos = mul(unity_ObjectToWorld, v.vertex).xyz;
                return o;
            }
            
            fixed4 frag (v2f i) : SV_Target
            {
                // 像素化处理
                float2 pixelUV = floor(i.uv * _PixelSize) / _PixelSize;
                fixed4 col = tex2D(_MainTex, pixelUV);
                
                // 颜色混合
                col.rgb *= _Color.rgb;
                
                // 透明度衰减
                float fade = 1.0 - (_Time.y * _FadeSpeed);
                col.a *= fade;
                
                return col;
            }
            ENDCG
        }
    }
}
```

// HotUpdateBuilder.cs
using System.IO;
using System.IO.Compression;

public class HotUpdateBuilder
{
    public static void BuildHotUpdate(string luaPath, string outputPath)
    {
        // 创建临时目录
        string tempDir = Path.Combine(Path.GetTempPath(), "HotUpdateTemp");
        Directory.CreateDirectory(tempDir);
        
        // 复制Lua文件
        foreach (string file in Directory.GetFiles(luaPath, "*.lua", SearchOption.AllDirectories))
        {
            string relativePath = Path.GetRelativePath(luaPath, file);
            string destPath = Path.Combine(tempDir, relativePath);
            Directory.CreateDirectory(Path.GetDirectoryName(destPath));
            File.Copy(file, destPath);
        }
        
        // 创建ZIP包
        ZipFile.CreateFromDirectory(tempDir, outputPath);
        
        // 清理临时文件
        Directory.Delete(tempDir, true);
    }
}

[UpdateInGroup(typeof(SimulationSystemGroup))]
public class AnimationSystem : SystemBase

    {
        float deltaTime = Time.DeltaTime;
        
        Entities.ForEach((ref AnimationComponent animation) => {
            animation.Progress += animation.Speed * deltaTime;

            {
                animation.Progress = 0.0f;
            }
        }).ScheduleParallel();

    protected override void OnUpdate()
    {
        float deltaTime = Time.DeltaTime;

            effect.ElapsedTime += deltaTime;
            if (effect.ElapsedTime >= effect.Duration)
            {
                EntityManager.DestroyEntity(entity);
            }
        }).ScheduleParallel();
    }
}

### 6.1 场景生成算法

```csharp
// ProceduralLevelGenerator.cs
using UnityEngine;
using System.Collections.Generic;

public class ProceduralLevelGenerator
{
    public static void GenerateLevel(int width, int height, float seed)
    {
        // 使用Perlin噪声生成地形
        float[,] noiseMap = new float[width, height];
        for (int x = 0; x < width; x++)
        {
            for (int y = 0; y < height; y++)
            {
                float noiseValue = Mathf.PerlinNoise(
                    x * 0.1f + seed, 
                    y * 0.1f + seed
                );
                
                // 根据噪声值生成地形
                if (noiseValue < 0.3f)
                {
                    CreateWater(x, y);
                }
                else if (noiseValue < 0.6f)
                {
                    CreateGrass(x, y);
                }
                else
                {
                    CreateRock(x, y);
                }
            }
        }
        
        // 生成路径
        GeneratePath(width, height);
    }
    
    private static void GeneratePath(int width, int height)
    {
        // 使用A*算法生成路径
        List<Vector2Int> path = AStar.FindPath(
            new Vector2Int(0, height / 2),
            new Vector2Int(width - 1, height / 2)
        );
        
        // 创建路径
        foreach (Vector2Int pos in path)
        {
            CreatePathTile(pos.x, pos.y);
        }
    }
}
```

public class PixelCharacterGenerator

    public static Sprite GenerateEnemySprite(int level)
    {
        // 基础像素网格 (16x16)
        Color32[] pixels = new Color32[16 * 16];

        Color bodyColor = new Color(

            Random.Range(0.2f, 0.6f),

        );

            {

            }
        }

        pixels[6 * 16 + 6] = Color.white;
        pixels[6 * 16 + 9] = Color.white;
        pixels[7 * 16 + 6] = Color.black;
        pixels[7 * 16 + 9] = Color.black;

// TowerSystem.cs
using UnityEngine;
using System.Collections.Generic;

public class TowerSystem : MonoBehaviour
{
    private Dictionary<int, TowerData> _towerDataDict = new Dictionary<int, TowerData>();
    
    public void Initialize()
    {
        // 加载塔配置
        LoadTowerData();
    }
    
    public TowerEntity CreateTower(int towerId, Vector3 position)
    {
        if (!_towerDataDict.TryGetValue(towerId, out TowerData data))
        {
            Debug.LogError($"Tower data not found: {towerId}");
            return null;
        }
        
        // 创建塔实体
        TowerEntity tower = new TowerEntity();
        tower.Initialize(data, position);
        
        return tower;
    }
    
    private void LoadTowerData()
    {
        // 从配置文件加载塔数据
        TextAsset configFile = Resources.Load<TextAsset>("Configs/Towers");
        TowerConfig config = JsonUtility.FromJson<TowerConfig>(configFile.text);
        
        foreach (TowerData data in config.towers)
        {
            _towerDataDict[data.id] = data;
        }
    }
}

// RoguelikeSystem.cs
using UnityEngine;
using System.Collections.Generic;

public class RoguelikeSystem : MonoBehaviour
{
    private Dictionary<string, float> _permanentUpgrades = new Dictionary<string, float>();
    
    public void ApplyPermanentUpgrade(string upgradeId, float value)
    {
        if (_permanentUpgrades.ContainsKey(upgradeId))
        {
            _permanentUpgrades[upgradeId] += value;
        }
        else
        {
            _permanentUpgrades[upgradeId] = value;
        }
        
        // 保存到PlayerPrefs
        PlayerPrefs.SetString("PermanentUpgrades", JsonUtility.ToJson(_permanentUpgrades));
    }
    
    public float GetUpgradeValue(string upgradeId)
    {
        _permanentUpgrades.TryGetValue(upgradeId, out float value);
        return value;
    }
    
    public List<UpgradeOption> GenerateRandomUpgrades()
    {
        List<UpgradeOption> options = new List<UpgradeOption>();
        
        // 随机生成3个升级选项
        for (int i = 0; i < 3; i++)
        {
            UpgradeOption option = new UpgradeOption();
            option.id = Random.Range(1, 10);
            option.name = GetUpgradeName(option.id);
            option.description = GetUpgradeDescription(option.id);
            option.value = Random.Range(0.1f, 0.5f);
            
            options.Add(option);
        }
        
        return options;
    }
}
